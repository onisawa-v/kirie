import cv2
import numpy as np

def extract_image(img_name, depth_name, threshold_lower, threshold_upper):

    img = cv2.imread(img_name)
    depth_map = cv2.imread(depth_name, cv2.IMREAD_GRAYSCALE)
    
    img = cv2.cvtColor(img, cv2.COLOR_RGB2RGBA)
    
    depth_map = cv2.normalize(depth_map, None, 0, 255, cv2.NORM_MINMAX)
    
    mask = cv2.inRange(depth_map, threshold_lower, threshold_upper)
    
    img[mask == 0] = (0, 0, 0, 0)

    cv2.imwrite(f"{img_name.split('.')[0]}_{threshold_lower}-{threshold_upper}.png", img)

extract_image("input.png", "input_depth.png", 0, 160)
